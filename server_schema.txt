DROP TABLE users;
CREATE TABLE users (
    user_id     serial CONSTRAINT firstkey PRIMARY KEY,
    fb_id       text,
    first_name  text NOT NULL,
    last_name   text NOT NULL,
    email       text UNIQUE NOT NULL ,
    password    text
);
CREATE INDEX ON users (user_id);
